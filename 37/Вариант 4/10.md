class DynamicProgramming:   
    # 1. Числа Фибоначчи
    def fibonacci_naive(self, n: int) -> int:
        if n <= 1:
            return n
        return self.fibonacci_naive(n-1) + self.fibonacci_naive(n-2)
    
    def fibonacci_memoization(self, n: int) -> int:
        memo = {}
        
        def fib_memo(k):
            if k in memo:
                return memo[k]
            if k <= 1:
                return k
            memo[k] = fib_memo(k-1) + fib_memo(k-2)
            return memo[k]
        
        return fib_memo(n)
    
    def fibonacci_tabulation(self, n: int) -> int:
        if n <= 1:
            return n
        
        dp = [0] * (n + 1)
        dp[1] = 1
        
        for i in range(2, n + 1):
            dp[i] = dp[i-1] + dp[i-2]
        
        return dp[n]
    
    def fibonacci_optimized(self, n: int) -> int:
        if n <= 1:
            return n
        
        prev, curr = 0, 1
        for _ in range(2, n + 1):
            prev, curr = curr, prev + curr
        
        return curr
    
    # 2. Задача о рюкзаке
    def knapsack_01(self, weights: List[int], values: List[int], capacity: int) -> int:
        n = len(weights)
        dp = [[0] * (capacity + 1) for _ in range(n + 1)]
        
        for i in range(1, n + 1):
            for w in range(1, capacity + 1):
                if weights[i-1] <= w:
                    dp[i][w] = max(
                        dp[i-1][w], 
                        values[i-1] + dp[i-1][w - weights[i-1]]
                    )
                else:
                    dp[i][w] = dp[i-1][w]
        
        selected_items = []
        w = capacity
        for i in range(n, 0, -1):
            if dp[i][w] != dp[i-1][w]:
                selected_items.append(i-1)
                w -= weights[i-1]
        
        return dp[n][capacity], selected_items[::-1]
    
    def knapsack_unbounded(self, weights: List[int], values: List[int], capacity: int) -> int:
        dp = [0] * (capacity + 1)
        
        for w in range(1, capacity + 1):
            for i in range(len(weights)):
                if weights[i] <= w:
                    dp[w] = max(dp[w], dp[w - weights[i]] + values[i])
        
        return dp[capacity]
    
    # 3. Наибольшая общая подпоследовательность (LCS)
    def longest_common_subsequence(self, text1: str, text2: str) -> Tuple[int, str]:
        m, n = len(text1), len(text2)
        dp = [[0] * (n + 1) for _ in range(m + 1)]

        for i in range(1, m + 1):
            for j in range(1, n + 1):
                if text1[i-1] == text2[j-1]:
                    dp[i][j] = dp[i-1][j-1] + 1
                else:
                    dp[i][j] = max(dp[i-1][j], dp[i][j-1])
        
        lcs = []
        i, j = m, n
        while i > 0 and j > 0:
            if text1[i-1] == text2[j-1]:
                lcs.append(text1[i-1])
                i -= 1
                j -= 1
            elif dp[i-1][j] > dp[i][j-1]:
                i -= 1
            else:
                j -= 1
        
        return dp[m][n], ''.join(reversed(lcs))
    
    # 4. Наибольшая возрастающая подпоследовательность (LIS)
    def longest_increasing_subsequence(self, nums: List[int]) -> Tuple[int, List[int]]:
        if not nums:
            return 0, []
        
        n = len(nums)
        dp = [1] * n 
        prev = [-1] * n 
        
        for i in range(n):
            for j in range(i):
                if nums[j] < nums[i] and dp[j] + 1 > dp[i]:
                    dp[i] = dp[j] + 1
                    prev[i] = j
        
        max_length = max(dp)
        max_index = dp.index(max_length)
        
        lis = []
        while max_index != -1:
            lis.append(nums[max_index])
            max_index = prev[max_index]
        
        return max_length, lis[::-1]
    
    def lis_nlogn(self, nums: List[int]) -> int:
        import bisect
        
        tails = []
        for num in nums:
            pos = bisect.bisect_left(tails, num)
            if pos == len(tails):
                tails.append(num)
            else:
                tails[pos] = num
        return len(tails)
    
    # 5. Размен монет
    def coin_change(self, coins: List[int], amount: int) -> int:
        dp = [float('inf')] * (amount + 1)
        dp[0] = 0
        
        for i in range(1, amount + 1):
            for coin in coins:
                if i >= coin:
                    dp[i] = min(dp[i], dp[i - coin] + 1)
        
        return dp[amount] if dp[amount] != float('inf') else -1
    
    def coin_change_ways(self, coins: List[int], amount: int) -> int:
        dp = [0] * (amount + 1)
        dp[0] = 1
        
        for coin in coins:
            for i in range(coin, amount + 1):
                dp[i] += dp[i - coin]
        
        return dp[amount]
    
    # 6. Расстояние Левенштейна
    def levenshtein_distance(self, word1: str, word2: str) -> int:
        m, n = len(word1), len(word2)
        dp = [[0] * (n + 1) for _ in range(m + 1)]
        
        for i in range(m + 1):
            dp[i][0] = i
        for j in range(n + 1):
            dp[0][j] = j
        
        for i in range(1, m + 1):
            for j in range(1, n + 1):
                if word1[i-1] == word2[j-1]:
                    cost = 0
                else:
                    cost = 1
                
                dp[i][j] = min(
                    dp[i-1][j] + 1,
                    dp[i][j-1] + 1,
                    dp[i-1][j-1] + cost
                )
        
        return dp[m][n]
    
    # 7. Оптимальное умножение матриц
    def matrix_chain_multiplication(self, dims: List[int]) -> Tuple[int, str]:
        n = len(dims) - 1
        dp = [[0] * n for _ in range(n)]
        split = [[0] * n for _ in range(n)]
        
        for length in range(2, n + 1):
            for i in range(n - length + 1):
                j = i + length - 1
                dp[i][j] = float('inf')
                
                for k in range(i, j):
                    cost = dp[i][k] + dp[k+1][j] + dims[i] * dims[k+1] * dims[j+1]
                    if cost < dp[i][j]:
                        dp[i][j] = cost
                        split[i][j] = k
        def build_order(i, j):
            if i == j:
                return f"A{i+1}"
            else:
                k = split[i][j]
                left = build_order(i, k)
                right = build_order(k+1, j)
                return f"({left} × {right})"
        
        return dp[0][n-1], build_order(0, n-1)
    
    # 8. Задача о наибольшей сумме подмассива (Kadane)
    def max_subarray_sum(self, nums: List[int]) -> Tuple[int, List[int]]:
        max_sum = nums[0]
        current_sum = nums[0]
        start = end = temp_start = 0
        
        for i in range(1, len(nums)):
            if nums[i] > current_sum + nums[i]:
                current_sum = nums[i]
                temp_start = i
            else:
                current_sum += nums[i]
            
            if current_sum > max_sum:
                max_sum = current_sum
                start = temp_start
                end = i
        
        return max_sum, nums[start:end+1]
    
    # 9. Задача о разбиении на подмножества равной суммы
    def can_partition(self, nums: List[int]) -> bool:
        total = sum(nums)
        if total % 2 != 0:
            return False
        
        target = total // 2
        dp = [False] * (target + 1)
        dp[0] = True
        
        for num in nums:
            for i in range(target, num - 1, -1):
                dp[i] = dp[i] or dp[i - num]
        
        return dp[target]
    
    # 10. Задача о редактировании с ограничениями
    def edit_distance_constrained(self, word1: str, word2: str, max_ops: int) -> bool:
        m, n = len(word1), len(word2)
        dp = [[0] * (n + 1) for _ in range(m + 1)]
        
        for i in range(m + 1):
            dp[i][0] = i
        for j in range(n + 1):
            dp[0][j] = j
        
        for i in range(1, m + 1):
            for j in range(1, n + 1):
                if word1[i-1] == word2[j-1]:
                    dp[i][j] = dp[i-1][j-1]
                else:
                    dp[i][j] = 1 + min(dp[i-1][j], dp[i][j-1], dp[i-1][j-1])
        
        return dp[m][n] <= max_ops
